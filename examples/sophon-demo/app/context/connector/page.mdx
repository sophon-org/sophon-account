# SophonWagmiConnector

The `SophonWagmiConnector` component integrates your Dynamic project settings with Wagmi

### Usage

```tsx
import { SophonContextProvider } from "@sophon-labs/account-react";
import { SophonWagmiConnector } from "@sophon-labs/account-wagmi";

function App() {
  return (
    <SophonContextProvider settings={{ partnerId: "your_partner_id" }}>
      <SophonWagmiConnector>
        <YourRoutes />
      </SophonWagmiConnector>
    </SophonContextProvider>
  );
}
```
